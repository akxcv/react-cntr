    AnimatedCounterDigit

    Extends:

    react~React.Component → AbstractCounterDigit → AnimatedCounterDigit

    animated digit component used when easing function is provided

    Example:

    <AnimatedCounterDigit
  digit='5'
  height={18}
  radix={10}
  digitMap={{}}
  digitWrapper={(digit) => digit}
  maxValue={9}
  easingFunction='ease-in'
  easingDuration={200}
/>
